425  template <typename U>
426  void draw_circle(const basic_point<U>& position, const float radius) noexcept
427  {
428  using value_t = typename basic_point<U>::value_type;
429 
430  auto error = -radius;
431  auto x = radius - 0.5f;
432  auto y = 0.5f;
433 
434  const auto cx = static_cast<float>(position.x()) - 0.5f;
435  const auto cy = static_cast<float>(position.y()) - 0.5f;
436 
437  while (x >= y) {
438  draw_point<value_t>({static_cast<value_t>(cx + x), static_cast<value_t>(cy + y)});
439  draw_point<value_t>({static_cast<value_t>(cx + y), static_cast<value_t>(cy + x)});
440 
441  if (x != 0) {
442  draw_point<value_t>({static_cast<value_t>(cx - x), static_cast<value_t>(cy + y)});
443  draw_point<value_t>({static_cast<value_t>(cx + y), static_cast<value_t>(cy - x)});
444  }
445 
446  if (y != 0) {
447  draw_point<value_t>({static_cast<value_t>(cx + x), static_cast<value_t>(cy - y)});
448  draw_point<value_t>({static_cast<value_t>(cx - y), static_cast<value_t>(cy + x)});
449  }
450 
451  if (x != 0 && y != 0) {
452  draw_point<value_t>({static_cast<value_t>(cx - x), static_cast<value_t>(cy - y)});
453  draw_point<value_t>({static_cast<value_t>(cx - y), static_cast<value_t>(cy - x)});
454  }
455 
456  error += y;
457  ++y;
458  error += y;
459 
460  if (error >= 0) {
461  --x;
462  error -= x;
463  error -= x;
464  }
465  }
466  }
